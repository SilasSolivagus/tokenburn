# tokenburn v0.1 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a CLI tool that intercepts AI tool API calls via local proxy, stores token usage in SQLite, and surfaces cost reports + waste detection.

**Architecture:** Local HTTP proxy on :10811 intercepts Anthropic/OpenAI API calls, extracts token usage from responses (including SSE streams), writes standardized `RequestRecord` rows to SQLite. CLI commands query the database for reports and waste pattern analysis.

**Tech Stack:** TypeScript, Node.js, better-sqlite3, commander, chalk, cli-table3, http-proxy, blessed (for live TUI)

```typescript
/**
 * Resolve a model ID to its canonical pricing key.
 * Strips date suffixes (e.g., "claude-opus-4-20250514" → "claude-opus-4")
 * and tries progressively shorter prefixes.
 */
export function resolvePricingKey(model: string): string | undefined {
  if (PRICING[model]) return model

  // Strip date suffix: claude-opus-4-20250514 → claude-opus-4
  const withoutDate = model.replace(/-\d{8}$/, '')
  if (PRICING[withoutDate]) return withoutDate

  // Try dropping last segment: claude-opus-4-6 already in table
  // but claude-haiku-4-5-20251001 → claude-haiku-4-5
  const segments = withoutDate.split('-')
  for (let i = segments.length; i >= 2; i--) {
    const key = segments.slice(0, i).join('-')
    if (PRICING[key]) return key
  }

  return undefined
}

export function getModelPricing(model: string): ModelPricing | undefined {
  const key = resolvePricingKey(model)
  return key ? PRICING[key] : undefined
}

export function isExpensiveModel(model: string): boolean {
  const pricing = getModelPricing(model)
  if (!pricing) return false
  return pricing.outputPerMillion >= 10
}
```

```typescript
export interface TokenCounts {
  inputTokens: number
  outputTokens: number
  cacheReadTokens: number
  cacheWriteTokens: number
}

export function calculateCost(model: string, tokens: TokenCounts): number {
  const pricing = getModelPricing(model)
  if (!pricing) return 0

  return (
    (tokens.inputTokens * pricing.inputPerMillion) / 1_000_000 +
    (tokens.outputTokens * pricing.outputPerMillion) / 1_000_000 +
    (tokens.cacheReadTokens * pricing.cacheReadPerMillion) / 1_000_000 +
    (tokens.cacheWriteTokens * pricing.cacheWritePerMillion) / 1_000_000
  )
}
```

```typescript
export interface RequestRecord {
  id: string
  timestamp: number
  provider: 'anthropic' | 'openai'
  model: string
  source: string
  inputTokens: number
  outputTokens: number
  cacheReadTokens: number

  durationMs: number
  promptHash: string
  toolUse: string // JSON stringified string[]
  stopReason: string

export const CREATE_TABLE_SQL = `
  CREATE TABLE IF NOT EXISTS requests (
    id TEXT PRIMARY KEY,
    timestamp INTEGER NOT NULL,
    provider TEXT NOT NULL,
    model TEXT NOT NULL,
    source TEXT NOT NULL DEFAULT 'unknown',
    inputTokens INTEGER NOT NULL DEFAULT 0,
    outputTokens INTEGER NOT NULL DEFAULT 0,
    cacheReadTokens INTEGER NOT NULL DEFAULT 0,
    cacheWriteTokens INTEGER NOT NULL DEFAULT 0,
    costUSD REAL NOT NULL DEFAULT 0,
    durationMs INTEGER NOT NULL DEFAULT 0,
    promptHash TEXT NOT NULL DEFAULT '',
    toolUse TEXT NOT NULL DEFAULT '[]',
    stopReason TEXT NOT NULL DEFAULT ''
```

- [ ] **Step 4: Create database helpers**

Create `src/db/db.ts`:
```typescript
import Database from 'better-sqlite3'
import path from 'path'
import os from 'os'
import { CREATE_TABLE_SQL, type RequestRecord } from './schema.js'

let db: Database.Database | null = null

export function getDbPath(): string {
  return path.join(os.homedir(), '.tokenburn', 'tokenburn.db')
}

export function getDb(dbPath?: string): Database.Database {
  if (db) return db
  const resolvedPath = dbPath ?? getDbPath()
  const dir = path.dirname(resolvedPath)
  const fs = await import('fs')
  if (!fs.existsSync(dir)) fs.mkdirSync(dir, { recursive: true })
  db = new Database(resolvedPath)
  db.pragma('journal_mode = WAL')
  db.exec(CREATE_TABLE_SQL)
  return db
}

export function closeDb(): void {
  if (db) {
    db.close()
    db = null
  }
}

export function insertRequest(record: RequestRecord): void {
  const d = getDb()
  const stmt = d.prepare(`
    INSERT OR REPLACE INTO requests
    (id, timestamp, provider, model, source, inputTokens, outputTokens,
     cacheReadTokens, cacheWriteTokens, costUSD, durationMs, promptHash, toolUse, stopReason)
    VALUES
    (@id, @timestamp, @provider, @model, @source, @inputTokens, @outputTokens,
     @cacheReadTokens, @cacheWriteTokens, @costUSD, @durationMs, @promptHash, @toolUse, @stopReason)
  `)
  stmt.run(record)
}

export interface QueryFilter {
  since?: number
  until?: number
  provider?: string
  model?: string
  source?: string
}

export function queryRequests(filter: QueryFilter): RequestRecord[] {
  const d = getDb()
  const conditions: string[] = []
  const params: Record<string, unknown> = {}

  if (filter.since !== undefined) {
    conditions.push('timestamp >= @since')
    params.since = filter.since
  }
  if (filter.until !== undefined) {
    conditions.push('timestamp <= @until')
    params.until = filter.until
  }
  if (filter.provider) {
    conditions.push('provider = @provider')
    params.provider = filter.provider
  }
  if (filter.model) {
    conditions.push('model = @model')
    params.model = filter.model
  }
  if (filter.source) {
    conditions.push('source = @source')
    params.source = filter.source
  }

  const where = conditions.length > 0 ? `WHERE ${conditions.join(' AND ')}` : ''
  return d.prepare(`SELECT * FROM requests ${where} ORDER BY timestamp ASC`).all(params) as RequestRecord[]
}
```

```typescript
const PATTERNS: Array<[RegExp, string]> = [
  [/claude-code/i, 'claude-code'],
  [/aider/i, 'aider'],
  [/cursor/i, 'cursor'],
  [/openclaw/i, 'openclaw'],
  [/continue/i, 'continue'],
  [/copilot/i, 'copilot'],
  [/cline/i, 'cline'],
]

export function detectSource(userAgent: string | undefined): string {
  if (!userAgent) return 'unknown'
  for (const [pattern, name] of PATTERNS) {
    if (pattern.test(userAgent)) return name
  }
  return 'unknown'
}
```

```typescript
export interface ParsedResponse {
  model: string
  inputTokens: number
  outputTokens: number
  cacheReadTokens: number
  cacheWriteTokens: number
  stopReason: string
  toolUse: string[]
}

export function parseAnthropicResponse(body: Record<string, unknown>): ParsedResponse {
  const usage = body.usage as Record<string, number> | undefined
  const content = body.content as Array<Record<string, unknown>> | undefined

  const toolUse = (content ?? [])
    .filter((c) => c.type === 'tool_use')
    .map((c) => c.name as string)

  return {
    model: (body.model as string) ?? 'unknown',
    inputTokens: usage?.input_tokens ?? 0,
    outputTokens: usage?.output_tokens ?? 0,
    cacheReadTokens: usage?.cache_read_input_tokens ?? 0,
    cacheWriteTokens: usage?.cache_creation_input_tokens ?? 0,
    stopReason: (body.stop_reason as string) ?? '',
    toolUse,
  }
}

export function parseAnthropicSSE(raw: string): ParsedResponse | null {
  let model = 'unknown'
  let inputTokens = 0
  let outputTokens = 0
  let cacheReadTokens = 0
  let cacheWriteTokens = 0
  let stopReason = ''
  const toolUse: string[] = []

  const lines = raw.split('\n')
  for (let i = 0; i < lines.length; i++) {
    const line = lines[i]
    if (!line.startsWith('data: ')) continue

    let event: Record<string, unknown>
    try {
      event = JSON.parse(line.slice(6))
    } catch {
      continue
    }

    if (event.type === 'message_start') {
      const msg = event.message as Record<string, unknown>
      model = (msg.model as string) ?? model
      const usage = msg.usage as Record<string, number> | undefined
      if (usage) {
        inputTokens = usage.input_tokens ?? 0
        cacheReadTokens = usage.cache_read_input_tokens ?? 0
        cacheWriteTokens = usage.cache_creation_input_tokens ?? 0
      }
    }

    if (event.type === 'content_block_start') {
      const cb = event.content_block as Record<string, unknown> | undefined
      if (cb?.type === 'tool_use') {
        toolUse.push(cb.name as string)
      }
    }

    if (event.type === 'message_delta') {
      const delta = event.delta as Record<string, unknown> | undefined
      stopReason = (delta?.stop_reason as string) ?? stopReason
      const usage = event.usage as Record<string, number> | undefined
      if (usage) {
        outputTokens = usage.output_tokens ?? outputTokens
      }
    }
  }

  if (model === 'unknown' && inputTokens === 0) return null
  return { model, inputTokens, outputTokens, cacheReadTokens, cacheWriteTokens, stopReason, toolUse }
}
```

```typescript
import type { ParsedResponse } from './anthropic.js'

export function parseOpenAIResponse(body: Record<string, unknown>): ParsedResponse {
  const usage = body.usage as Record<string, unknown> | undefined
  const choices = body.choices as Array<Record<string, unknown>> | undefined
  const firstChoice = choices?.[0]
  const message = firstChoice?.message as Record<string, unknown> | undefined
  const toolCalls = message?.tool_calls as Array<Record<string, unknown>> | undefined

  const promptDetails = usage?.prompt_tokens_details as Record<string, number> | undefined

  const toolUse = (toolCalls ?? []).map((tc) => {
    const fn = tc.function as Record<string, unknown>
    return fn.name as string
  })

  return {
    model: (body.model as string) ?? 'unknown',
    inputTokens: (usage?.prompt_tokens as number) ?? 0,
    outputTokens: (usage?.completion_tokens as number) ?? 0,
    cacheReadTokens: promptDetails?.cached_tokens ?? 0,
    cacheWriteTokens: 0,
    stopReason: (firstChoice?.finish_reason as string) ?? '',
    toolUse,
  }
}

export function parseOpenAISSE(raw: string): ParsedResponse | null {
  let model = 'unknown'
  let inputTokens = 0
  let outputTokens = 0
  let cacheReadTokens = 0
  let stopReason = ''
  const toolUse: string[] = []

  const lines = raw.split('\n')
  for (const line of lines) {
    if (!line.startsWith('data: ') || line === 'data: [DONE]') continue

    let event: Record<string, unknown>
    try {
      event = JSON.parse(line.slice(6))
    } catch {
      continue
    }

    model = (event.model as string) ?? model

    const choices = event.choices as Array<Record<string, unknown>> | undefined
    if (choices?.[0]) {
      const choice = choices[0]
      if (choice.finish_reason) {
        stopReason = choice.finish_reason as string
      }
      const delta = choice.delta as Record<string, unknown> | undefined
      const tc = delta?.tool_calls as Array<Record<string, unknown>> | undefined
      if (tc) {
        for (const t of tc) {
          const fn = t.function as Record<string, unknown> | undefined
          if (fn?.name) toolUse.push(fn.name as string)
        }
      }
    }

    const usage = event.usage as Record<string, number> | undefined
    if (usage) {
      inputTokens = usage.prompt_tokens ?? inputTokens
      outputTokens = usage.completion_tokens ?? outputTokens
      const details = (event.usage as Record<string, unknown>)?.prompt_tokens_details as Record<string, number> | undefined
      cacheReadTokens = details?.cached_tokens ?? cacheReadTokens
    }
  }

  if (model === 'unknown' && inputTokens === 0) return null
  return { model, inputTokens, outputTokens, cacheReadTokens, cacheWriteTokens: 0, stopReason, toolUse }
}
```

```typescript
import crypto from 'crypto'
import type { ParsedResponse } from './parsers/anthropic.js'
import { detectSource } from './source-detect.js'
import { calculateCost } from '../pricing/cost.js'
import { insertRequest } from '../db/db.js'

export interface CollectInput {
  provider: 'anthropic' | 'openai'
  requestBody: string
  parsed: ParsedResponse
  userAgent: string | undefined
  durationMs: number
}

export function collectRequest(input: CollectInput): void {
  const promptHash = crypto
    .createHash('sha256')
    .update(input.requestBody)
    .digest('hex')
    .slice(0, 16)

  const costUSD = calculateCost(input.parsed.model, {
    inputTokens: input.parsed.inputTokens,
    outputTokens: input.parsed.outputTokens,
    cacheReadTokens: input.parsed.cacheReadTokens,
    cacheWriteTokens: input.parsed.cacheWriteTokens,
  })

  insertRequest({
    id: crypto.randomUUID(),
    timestamp: Date.now(),
    provider: input.provider,
    model: input.parsed.model,
    source: detectSource(input.userAgent),
    inputTokens: input.parsed.inputTokens,
    outputTokens: input.parsed.outputTokens,
    cacheReadTokens: input.parsed.cacheReadTokens,
    cacheWriteTokens: input.parsed.cacheWriteTokens,
    costUSD,
    durationMs: input.durationMs,
    promptHash,
    toolUse: JSON.stringify(input.parsed.toolUse),
    stopReason: input.parsed.stopReason,
  })
}
```

```typescript
import http from 'http'
import { parseAnthropicResponse, parseAnthropicSSE } from './parsers/anthropic.js'
import { parseOpenAIResponse, parseOpenAISSE } from './parsers/openai.js'
import { collectRequest } from './collector.js'

const DEFAULT_TARGETS = {
  anthropic: 'https://api.anthropic.com',
  openai: 'https://api.openai.com',
}

interface ProxyConfig {
  port?: number
  targets?: Partial<typeof DEFAULT_TARGETS>
}

let server: http.Server | null = null

export async function startProxy(config: ProxyConfig = {}): Promise<void> {
  const port = config.port ?? 10811
  const targets = { ...DEFAULT_TARGETS, ...config.targets }

  server = http.createServer(async (clientReq, clientRes) => {
    const startTime = Date.now()

    // Determine provider from URL prefix
    let provider: 'anthropic' | 'openai' | null = null
    let targetBase = ''
    let strippedPath = clientReq.url ?? '/'

    if (strippedPath.startsWith('/anthropic')) {
      provider = 'anthropic'
      targetBase = targets.anthropic
      strippedPath = strippedPath.slice('/anthropic'.length)
    } else if (strippedPath.startsWith('/openai')) {
      provider = 'openai'
      targetBase = targets.openai
      strippedPath = strippedPath.slice('/openai'.length)
    }

    if (!provider) {
      clientRes.writeHead(404, { 'Content-Type': 'application/json' })
      clientRes.end(JSON.stringify({ error: 'Unknown provider. Use /anthropic/* or /openai/*' }))
      return
    }

    // Read the full request body
    const requestChunks: Buffer[] = []
    for await (const chunk of clientReq) {
      requestChunks.push(chunk as Buffer)
    }
    const requestBody = Buffer.concat(requestChunks).toString('utf8')

    // Forward the request
    const targetUrl = new URL(strippedPath, targetBase)
    const headers: Record<string, string> = {}
    for (const [key, value] of Object.entries(clientReq.headers)) {
      if (key === 'host') continue
      if (value) headers[key] = Array.isArray(value) ? value[0] : value
    }

    try {
      const upstreamRes = await fetch(targetUrl.toString(), {
        method: clientReq.method ?? 'POST',
        headers,
        body: requestBody,
      })

      // Forward response headers
      const resHeaders: Record<string, string> = {}
      upstreamRes.headers.forEach((value, key) => {
        resHeaders[key] = value
      })
      clientRes.writeHead(upstreamRes.status, resHeaders)

      // Read + forward response body
      const responseChunks: Buffer[] = []
      const reader = upstreamRes.body?.getReader()
      if (reader) {
        while (true) {
          const { done, value } = await reader.read()
          if (done) break
          responseChunks.push(Buffer.from(value))
          clientRes.write(value)
        }
      }
      clientRes.end()

      const responseBody = Buffer.concat(responseChunks).toString('utf8')
      const durationMs = Date.now() - startTime
      const isStreaming = resHeaders['content-type']?.includes('text/event-stream')
      const userAgent = clientReq.headers['user-agent']

      // Parse and collect in background (don't block response)
      try {
        let parsed
        if (provider === 'anthropic') {
          parsed = isStreaming
            ? parseAnthropicSSE(responseBody)
            : parseAnthropicResponse(JSON.parse(responseBody))
        } else {
          parsed = isStreaming
            ? parseOpenAISSE(responseBody)
            : parseOpenAIResponse(JSON.parse(responseBody))
        }

        if (parsed) {
          collectRequest({ provider, requestBody, parsed, userAgent, durationMs })
        }
      } catch {
        // Parse failure should never break the proxy
      }
    } catch (err) {
      // Upstream failure: return 502
      clientRes.writeHead(502, { 'Content-Type': 'application/json' })
      clientRes.end(JSON.stringify({ error: 'Upstream request failed', detail: String(err) }))
    }
  })

  await new Promise<void>((resolve) => server!.listen(port, resolve))
}

export async function stopProxy(): Promise<void> {
  if (!server) return
  await new Promise<void>((resolve) => server!.close(() => resolve()))
  server = null
}

export function isProxyRunning(): boolean {
  return server !== null
}

export function getProxyPort(): number {
  const addr = server?.address()
  if (typeof addr === 'object' && addr) return addr.port
  return 0
}
```

```typescript
import { spawn } from 'child_process'
import fs from 'fs'
import path from 'path'
import os from 'os'

const PID_DIR = path.join(os.homedir(), '.tokenburn')
const PID_FILE = path.join(PID_DIR, 'proxy.pid')
const LOG_FILE = path.join(PID_DIR, 'proxy.log')

export function daemonStart(scriptPath: string, port: number): void {
  if (isDaemonRunning()) {
    console.log(`Proxy already running (PID ${readPid()})`)
    return
  }

  if (!fs.existsSync(PID_DIR)) fs.mkdirSync(PID_DIR, { recursive: true })

  const logFd = fs.openSync(LOG_FILE, 'a')
  const child = spawn(process.execPath, [scriptPath, '--daemon-child', '--port', String(port)], {
    detached: true,
    stdio: ['ignore', logFd, logFd],
    env: { ...process.env, TOKENBURN_DAEMON: '1' },
  })

  child.unref()
  fs.writeFileSync(PID_FILE, String(child.pid))
  fs.closeSync(logFd)
  console.log(`Proxy started in background (PID ${child.pid}, port ${port})`)
  console.log(`Logs: ${LOG_FILE}`)
}

export function daemonStop(): void {
  const pid = readPid()
  if (!pid) {
    console.log('No running proxy found')
    return
  }

  try {
    process.kill(pid, 'SIGTERM')
    fs.unlinkSync(PID_FILE)
    console.log(`Proxy stopped (PID ${pid})`)
  } catch {
    fs.unlinkSync(PID_FILE)
    console.log('Proxy was not running, cleaned up stale PID file')
  }
}

export function isDaemonRunning(): boolean {
  const pid = readPid()
  if (!pid) return false
  try {
    process.kill(pid, 0)
    return true
  } catch {
    // Stale PID file
    try { fs.unlinkSync(PID_FILE) } catch {}
    return false
  }
}

function readPid(): number | null {
  try {
    const content = fs.readFileSync(PID_FILE, 'utf8').trim()
    const pid = parseInt(content, 10)
    return isNaN(pid) ? null : pid
  } catch {
    return null
  }
}

export function getDaemonStatus(): { running: boolean; pid: number | null; logFile: string } {
  const pid = readPid()
  return { running: isDaemonRunning(), pid, logFile: LOG_FILE }
}
```

```typescript
import { getDb } from '../db/db.js'
import type { QueryFilter } from '../db/db.js'

function buildWhere(filter: QueryFilter): { clause: string; params: Record<string, unknown> } {
  const conditions: string[] = []
  const params: Record<string, unknown> = {}
  if (filter.since !== undefined) { conditions.push('timestamp >= @since'); params.since = filter.since }
  if (filter.until !== undefined) { conditions.push('timestamp <= @until'); params.until = filter.until }
  if (filter.provider) { conditions.push('provider = @provider'); params.provider = filter.provider }
  if (filter.model) { conditions.push('model = @model'); params.model = filter.model }
  if (filter.source) { conditions.push('source = @source'); params.source = filter.source }
  const clause = conditions.length > 0 ? `WHERE ${conditions.join(' AND ')}` : ''
  return { clause, params }
}

export interface AggregateRow {
  key: string
  model?: string
  source?: string
  day?: string
  hour?: number
  totalCost: number
  totalInput: number
  totalOutput: number
  requestCount: number
}

export function aggregateByModel(filter: QueryFilter): (AggregateRow & { model: string })[] {
  const db = getDb()
  const { clause, params } = buildWhere(filter)
  return db.prepare(`
    SELECT model, SUM(costUSD) as totalCost, SUM(inputTokens) as totalInput,
           SUM(outputTokens) as totalOutput, COUNT(*) as requestCount
    FROM requests ${clause}
    GROUP BY model ORDER BY totalCost DESC
  `).all(params) as (AggregateRow & { model: string })[]
}

export function aggregateBySource(filter: QueryFilter): (AggregateRow & { source: string })[] {
  const db = getDb()
  const { clause, params } = buildWhere(filter)
  return db.prepare(`
    SELECT source, SUM(costUSD) as totalCost, SUM(inputTokens) as totalInput,
           SUM(outputTokens) as totalOutput, COUNT(*) as requestCount
    FROM requests ${clause}
    GROUP BY source ORDER BY totalCost DESC
  `).all(params) as (AggregateRow & { source: string })[]
}

export function aggregateByDay(filter: QueryFilter): (AggregateRow & { day: string })[] {
  const db = getDb()
  const { clause, params } = buildWhere(filter)
  return db.prepare(`
    SELECT date(timestamp / 1000, 'unixepoch', 'localtime') as day,
           SUM(costUSD) as totalCost, SUM(inputTokens) as totalInput,
           SUM(outputTokens) as totalOutput, COUNT(*) as requestCount
    FROM requests ${clause}
    GROUP BY day ORDER BY day ASC
  `).all(params) as (AggregateRow & { day: string })[]
}

export function aggregateByHour(filter: QueryFilter): (AggregateRow & { hour: number })[] {
  const db = getDb()
  const { clause, params } = buildWhere(filter)
  return db.prepare(`
    SELECT CAST(strftime('%H', timestamp / 1000, 'unixepoch', 'localtime') AS INTEGER) as hour,
           SUM(costUSD) as totalCost, SUM(inputTokens) as totalInput,
           SUM(outputTokens) as totalOutput, COUNT(*) as requestCount
    FROM requests ${clause}
    GROUP BY hour ORDER BY hour ASC
  `).all(params) as (AggregateRow & { hour: number })[]

export interface Summary {
  totalCost: number
  totalRequests: number
  totalInputTokens: number
  totalOutputTokens: number
  totalCacheReadTokens: number
  avgCostPerRequest: number
}

export function summarize(filter: QueryFilter): Summary {
  const db = getDb()
  const { clause, params } = buildWhere(filter)
  const row = db.prepare(`
    SELECT COALESCE(SUM(costUSD), 0) as totalCost,
           COUNT(*) as totalRequests,
           COALESCE(SUM(inputTokens), 0) as totalInputTokens,
           COALESCE(SUM(outputTokens), 0) as totalOutputTokens,
           COALESCE(SUM(cacheReadTokens), 0) as totalCacheReadTokens
    FROM requests ${clause}
  `).get(params) as Record<string, number>

  return {
    totalCost: row.totalCost,
    totalRequests: row.totalRequests,
    totalInputTokens: row.totalInputTokens,
    totalOutputTokens: row.totalOutputTokens,
    totalCacheReadTokens: row.totalCacheReadTokens,
    avgCostPerRequest: row.totalRequests > 0 ? row.totalCost / row.totalRequests : 0,
  }
}
```

```typescript
    SELECT id, toolUse, costUSD FROM requests
    WHERE timestamp >= @since ORDER BY timestamp ASC
  `).all({ since: filter.since ?? 0 }) as Array<{ id: string; toolUse: string; costUSD: number }>

  let idleCost = 0
  let totalIdleCost = 0

  for (const row of rows) {
    const tools = JSON.parse(row.toolUse) as string[]
    if (tools.length === 0) {
      streak++
      idleCost += row.costUSD
    } else {

      idleCost = 0
```

```typescript
  })
  for (const row of rows) {
    table.push([row.model, formatCost(row.totalCost), row.requestCount, formatTokens(row.totalInput), formatTokens(row.totalOutput)])
  }
  return header('  By Model:\n') + table.toString() + '\n'
}

export function renderBySource(rows: (AggregateRow & { source: string })[]): string {
  if (rows.length === 0) return dim('  No data\n')
  const table = new Table({
    head: ['Source', 'Cost', 'Requests', 'Input', 'Output'],
    style: { head: ['cyan'] },
  })
  for (const row of rows) {
    table.push([row.source, formatCost(row.totalCost), row.requestCount, formatTokens(row.totalInput), formatTokens(row.totalOutput)])
  }
  return header('  By Source:\n') + table.toString() + '\n'
}

export function renderByDay(rows: (AggregateRow & { day: string })[]): string {
  if (rows.length === 0) return dim('  No data\n')
  const maxCost = Math.max(...rows.map((r) => r.totalCost))
  const lines = [header('  Daily spending:\n')]
  for (const row of rows) {
    const barLen = maxCost > 0 ? Math.round((row.totalCost / maxCost) * 30) : 0
    const bar = chalk.green('█'.repeat(barLen))
    lines.push(`  ${row.day}  ${bar} ${formatCost(row.totalCost)}`)
  }
  lines.push('')
  return lines.join('\n')
}

export function renderByHour(rows: (AggregateRow & { hour: number })[]): string {
  if (rows.length === 0) return dim('  No data\n')
  const maxCost = Math.max(...rows.map((r) => r.totalCost))
  const lines = [header('  Hourly distribution:\n')]
  for (let h = 0; h < 24; h++) {
    const row = rows.find((r) => r.hour === h)
    const cost = row?.totalCost ?? 0
    const barLen = maxCost > 0 ? Math.round((cost / maxCost) * 30) : 0
    const bar = chalk.green('█'.repeat(barLen))
    lines.push(`  ${String(h).padStart(2, '0')}:00  ${bar} ${cost > 0 ? formatCost(cost) : dim('-')}`)
  }
  lines.push('')
  return lines.join('\n')
}
```

```typescript
import chalk from 'chalk'
import { formatCost, severityIcon, header, dim } from './format.js'
import type { WasteDetection } from '../analyzer/rules/index.js'

export function renderScan(detections: WasteDetection[], totalCost: number): string {
  const lines = [
    '',
    header(`🔥 Total spending in period: ${formatCost(totalCost)}`),
    '',
  ]

  if (detections.length === 0) {
    lines.push(chalk.green('  ✨ No waste detected! Your AI spending looks efficient.'))
    lines.push('')
    return lines.join('\n')
  }

  lines.push(header(`  Waste detection (${detections.length} issues found):`))
  lines.push('')

  for (const d of detections) {
    lines.push(`  ${severityIcon(d.severity)} ${chalk.bold(d.message)} — ${formatCost(d.wastedUSD)} wasted`)
    lines.push(dim(`     ${d.detail}`))
    lines.push(chalk.cyan(`     → ${d.suggestion}`))
    lines.push('')
  }

  const totalSavable = detections.reduce((sum, d) => sum + d.savableUSD, 0)
  if (totalSavable > 0) {
    const pct = totalCost > 0 ? Math.round((totalSavable / totalCost) * 100) : 0
    lines.push(chalk.bold.green(`  💰 Potential savings: ${formatCost(totalSavable)}/period (${pct}%)`))
    lines.push('')
  }

  return lines.join('\n')
}
```

````markdown
```

**Zero data leaves your machine.** All analysis is local.
````
