```typescript
import type { QueryFilter } from '../../db/db.js'
import { duplicateRequests } from './duplicate-requests.js'
import { modelOveruse } from './model-overuse.js'
import { contextExplosion } from './context-explosion.js'

import { truncationWaste } from './truncation-waste.js'
import { idleChat } from './idle-chat.js'
import { lateNight } from './late-night.js'
import { retryStorm } from './retry-storm.js'
// v0.2 rules — will be imported as they are created in later tasks
// import { giantRequest } from './giant-request.js'
// ... (added incrementally)

export type Severity = 'high' | 'medium' | 'low' | 'info'

export interface WasteDetection {
  rule: string
  severity: Severity
  wastedUSD: number
  savableUSD: number
  message: string
  detail: string
  suggestion: string
  fix?: string  // executable fix (e.g., config snippet to add to CLAUDE.md)
}

export type RuleFn = (filter: QueryFilter) => WasteDetection | null

export interface RuleEntry {
  name: string
  description: string
  fn: RuleFn
}

const ALL_RULES: RuleEntry[] = [
  { name: 'duplicate-requests', description: 'Detect repeated identical prompts', fn: duplicateRequests },
  { name: 'model-overuse', description: 'Expensive models for short outputs', fn: modelOveruse },
  { name: 'context-explosion', description: 'Too many requests with >100k input tokens', fn: contextExplosion },
  { name: 'low-cache-hit', description: 'Cache hit rate below 10%', fn: lowCacheHit },
  { name: 'truncation-waste', description: 'Responses truncated by max_tokens', fn: truncationWaste },
  { name: 'idle-chat', description: '3+ turns without tool use', fn: idleChat },
  { name: 'late-night', description: 'Requests between 1-5am', fn: lateNight },
  { name: 'retry-storm', description: 'Rapid retries within 5 seconds', fn: retryStorm },
]

const SEVERITY_ORDER: Record<Severity, number> = { high: 0, medium: 1, low: 2, info: 3 }

export function listRules(): RuleEntry[] {
  return ALL_RULES
}

export function runAllRules(filter: QueryFilter, disabledRules: string[] = []): WasteDetection[] {
  const results: WasteDetection[] = []
  for (const entry of ALL_RULES) {
    if (disabledRules.includes(entry.name)) continue
    const detection = entry.fn(filter)
    if (detection !== null) results.push(detection)
  }
  return results.sort((a, b) => {
    const severityDiff = SEVERITY_ORDER[a.severity] - SEVERITY_ORDER[b.severity]
    if (severityDiff !== 0) return severityDiff
    return b.wastedUSD - a.wastedUSD
  })
}
```

```typescript
import { getDb, type QueryFilter } from '../../db/db.js'
import { buildWhere } from '../analyzer.js'
import type { RuleFn, WasteDetection } from './index.js'

export const giantRequest: RuleFn = (filter: QueryFilter): WasteDetection | null => {
  const db = getDb()
  const { sql, params } = buildWhere(filter)
  const rows = db.prepare(`
    SELECT COUNT(*) as cnt, SUM(costUSD) as totalCost, MAX(costUSD) as maxCost
    FROM requests ${sql} AND costUSD > 2
  `).get(params) as { cnt: number; totalCost: number; maxCost: number }
  if (rows.cnt === 0) return null
  return {
    rule: 'giant-request', severity: rows.totalCost > 10 ? 'high' : 'medium',
    wastedUSD: rows.totalCost * 0.3, savableUSD: rows.totalCost * 0.3,
    message: `${rows.cnt} giant request(s) costing >$2 each (max: $${rows.maxCost.toFixed(2)})`,
    detail: `${rows.cnt} requests individually cost over $2. Total: $${rows.totalCost.toFixed(2)}`,
    suggestion: 'Break large tasks into smaller steps to reduce per-request cost',
    fix: '# Add to CLAUDE.md:\nBreak complex tasks into smaller, focused steps. Avoid sending entire codebases as context.',
  }
}
```

```typescript
import { getDb, type QueryFilter } from '../../db/db.js'
import { buildWhere } from '../analyzer.js'
import type { RuleFn, WasteDetection } from './index.js'

export const longTailSession: RuleFn = (filter: QueryFilter): WasteDetection | null => {
  const db = getDb()
  const { sql, params } = buildWhere(filter)
  const sessions = db.prepare(`
    SELECT sessionId, MIN(timestamp) as startTs, MAX(timestamp) as endTs,
           COUNT(*) as cnt, SUM(costUSD) as totalCost
    FROM requests ${sql} AND sessionId != ''
    GROUP BY sessionId
    HAVING (endTs - startTs) > 7200000
  `).all(params) as Array<{ sessionId: string; startTs: number; endTs: number; cnt: number; totalCost: number }>
  if (sessions.length === 0) return null
  const totalWaste = sessions.reduce((s, r) => s + r.totalCost * 0.2, 0)
  return {
    rule: 'long-tail-session', severity: 'low',
    wastedUSD: totalWaste, savableUSD: totalWaste,
    message: `${sessions.length} session(s) ran longer than 2 hours`,
    detail: 'Long sessions accumulate large contexts, reducing efficiency over time',
    suggestion: 'Split long tasks into multiple sessions to reset context',
    fix: '# Tip: Start a new session every 1-2 hours. Use /clear to reset context when switching subtasks.',
```

```typescript
import { getDb, type QueryFilter } from '../../db/db.js'
import { buildWhere } from '../analyzer.js'
import type { RuleFn, WasteDetection } from './index.js'

export const rejectedGeneration: RuleFn = (filter: QueryFilter): WasteDetection | null => {
  const db = getDb()
  const { sql, params } = buildWhere(filter)
  const rows = db.prepare(`
    SELECT a.id, a.costUSD FROM requests a
    INNER JOIN requests b ON a.promptHash = b.promptHash
      AND a.id != b.id
      AND ABS(a.timestamp - b.timestamp) < 30000
      AND ABS(a.timestamp - b.timestamp) > 1000
    ${sql.replace('WHERE', 'AND')}
  `).all(params) as Array<{ id: string; costUSD: number }>
  if (rows.length < 2) return null
  const wastedUSD = rows.reduce((s, r) => s + r.costUSD, 0) / 2
  return {
    rule: 'rejected-generation', severity: 'medium',
    wastedUSD, savableUSD: wastedUSD,
    message: `~${Math.floor(rows.length / 2)} generation(s) appear to have been rejected and re-sent`,
    detail: 'Similar prompts sent within 30 seconds suggest rejected outputs',
    suggestion: 'Refine your prompt before sending rather than regenerating',
    fix: '# Tip: Edit your prompt to be more specific instead of hitting regenerate.',
  }
}
```

**provider-price-gap.ts:**
```typescript
import { getDb, type QueryFilter } from '../../db/db.js'
import { buildWhere } from '../analyzer.js'
import type { RuleFn, WasteDetection } from './index.js'

export const providerPriceGap: RuleFn = (filter: QueryFilter): WasteDetection | null => {
  const db = getDb()
  const { sql, params } = buildWhere(filter)
  const providers = db.prepare(`
    SELECT provider, SUM(costUSD) as totalCost, COUNT(*) as cnt
    FROM requests ${sql}
    GROUP BY provider HAVING cnt >= 5
  `).all(params) as Array<{ provider: string; totalCost: number; cnt: number }>
  if (providers.length < 2) return null
  const sorted = providers.sort((a, b) => (b.totalCost / b.cnt) - (a.totalCost / a.cnt))
  const expensive = sorted[0]
  const cheap = sorted[sorted.length - 1]
  const expAvg = expensive.totalCost / expensive.cnt
  const cheapAvg = cheap.totalCost / cheap.cnt
  if (cheapAvg === 0 || expAvg / cheapAvg < 1.5) return null
  return {
    rule: 'provider-price-gap', severity: 'info',
    wastedUSD: 0, savableUSD: expensive.totalCost - (expensive.cnt * cheapAvg),
    message: `${expensive.provider} costs ${(expAvg / cheapAvg).toFixed(1)}x more per request than ${cheap.provider}`,
    detail: `${expensive.provider}: avg $${expAvg.toFixed(3)}/req, ${cheap.provider}: avg $${cheapAvg.toFixed(3)}/req`,
    suggestion: `Consider using ${cheap.provider} for suitable tasks`,
    fix: `# Tip: For tasks where quality is comparable, use ${cheap.provider} to save ~$${(expensive.totalCost - expensive.cnt * cheapAvg).toFixed(2)}.`,
  }
}
```

```typescript
import fs from 'fs'
import path from 'path'
import readline from 'readline'

/**
 * Apply a fix string to CLAUDE.md in the current working directory.
 * If CLAUDE.md doesn't exist, create it.
 */
export function applyFixToClaudeMd(fix: string, cwd: string = process.cwd()): boolean {
  const claudeMdPath = path.join(cwd, 'CLAUDE.md')
  // Strip "# Add to CLAUDE.md:" prefix lines
  const content = fix
    .split('\n')
    .filter(line => !line.startsWith('# Add to CLAUDE.md:') && !line.startsWith('# Tip:'))
    .join('\n')
    .trim()

  if (!content) return false

  const existing = fs.existsSync(claudeMdPath) ? fs.readFileSync(claudeMdPath, 'utf8') : ''
  if (existing.includes(content)) return false // already applied

  const newContent = existing ? `${existing}\n\n${content}\n` : `${content}\n`
  fs.writeFileSync(claudeMdPath, newContent)
  return true
}

/**
 * Interactive prompt: ask user y/n/skip for each fix.
 */
export async function promptForFix(question: string): Promise<'y' | 'n' | 'skip'> {
  const rl = readline.createInterface({ input: process.stdin, output: process.stdout })
  return new Promise((resolve) => {
    rl.question(question, (answer) => {
      rl.close()
      const a = answer.trim().toLowerCase()
      if (a === 'y' || a === 'yes') resolve('y')
      else if (a === 'n' || a === 'no') resolve('n')
      else resolve('skip')
    })
  })
}
```

```typescript
function parsePeriod(period: string): number {
  const match = period.match(/^(\d+(?:\.\d+)?)([hdwm])$/)
  if (!match) throw new Error(`Invalid period: ${period}. Use formats like 7d, 24h, 2w, 1m.`)
  const value = parseFloat(match[1])
  const unit = match[2]
  const msPerUnit: Record<string, number> = {
    h: 60 * 60 * 1000, d: 24 * 60 * 60 * 1000,
    w: 7 * 24 * 60 * 60 * 1000, m: 30 * 24 * 60 * 60 * 1000,
  }
  return Date.now() - Math.round(value * msPerUnit[unit])
}

const SEVERITY_LEVELS: Severity[] = ['high', 'medium', 'low', 'info']

export const scanCommand = new Command('scan')
  .description('Scan for waste patterns in AI spending')
  .option('--last <period>', 'Time period (e.g. 7d, 24h)', '7d')
  .option('--severity <level>', 'Minimum severity: high, medium, low, info', 'info')
  .option('--fix', 'Interactive fix mode — apply fixes one by one', false)
  .option('--rules', 'List all available rules', false)
  .option('--disable <rules>', 'Comma-separated rules to skip')
  .option('--json', 'Output as JSON', false)
  .action(async (opts: { last: string; severity: string; fix: boolean; rules: boolean; disable?: string; json: boolean }) => {
    if (opts.rules) {
      const rules = listRules()
      console.log(chalk.bold(`\n  ${rules.length} rules available:\n`))
      for (const r of rules) {
        console.log(`  ${chalk.cyan(r.name.padEnd(25))} ${r.description}`)
      }
      console.log('')
      return
    }

    const since = parsePeriod(opts.last)
    const filter = { since }
    const disabled = opts.disable ? opts.disable.split(',').map(s => s.trim()) : []

    const minSeverity = opts.severity as Severity
    const minIndex = SEVERITY_LEVELS.indexOf(minSeverity)
    if (minIndex === -1) {
      console.error(`Invalid severity: ${opts.severity}. Choose from: high, medium, low, info`)
      process.exit(1)
    }

    const allDetections = runAllRules(filter, disabled)
    const detections = allDetections.filter(d => SEVERITY_LEVELS.indexOf(d.severity) <= minIndex)
    const summary = summarize(filter)

    if (opts.json) {
      console.log(JSON.stringify({ totalCost: summary.totalCost, detections }, null, 2))
      return
    }

    if (!opts.fix) {
      process.stdout.write(renderScan(detections, summary.totalCost))
      return
    }

    // Interactive fix mode
    console.log(renderScan(detections, summary.totalCost))
    let applied = 0
    for (const d of detections) {
      if (!d.fix) continue
      console.log(chalk.bold(`  Fix for: ${d.rule}`))
      console.log(chalk.dim(`  ${d.fix}`))
      const answer = await promptForFix(chalk.cyan('  Apply? [y/n/skip] '))
      if (answer === 'y') {
        const ok = applyFixToClaudeMd(d.fix)
        if (ok) { console.log(chalk.green('  ✓ Applied')); applied++ }
        else console.log(chalk.dim('  Already applied or empty fix'))
      }
      console.log('')
    }
    if (applied > 0) {
      console.log(chalk.bold.green(`  ${applied} fix(es) applied. Review: git diff CLAUDE.md`))
    }
  })
```

```typescript
import { getDb, type QueryFilter, queryRequests } from '../db/db.js'
import { calculateCost } from '../pricing/cost.js'
import { summarize } from './analyzer.js'
import { runAllRules } from './rules/index.js'

export interface OptimizationPlan {
  name: string
  description: string
  savingsUSD: number
  savingsPercent: number
  fix?: string
}

export interface SimulationResult {
  targetModel: string
  currentCost: number
  simulatedCost: number
  savings: number
  savingsPercent: number
  requestCount: number
  warning?: string
}

export function generateOptimizations(filter: QueryFilter): OptimizationPlan[] {
  const detections = runAllRules(filter)
  const summary = summarize(filter)
  const plans: OptimizationPlan[] = []

  for (const d of detections) {
    if (d.savableUSD > 0) {
      plans.push({
        name: d.rule,
        description: d.message,
        savingsUSD: d.savableUSD,
        savingsPercent: summary.totalCost > 0 ? (d.savableUSD / summary.totalCost) * 100 : 0,
        fix: d.fix,
      })
    }
  }

  return plans.sort((a, b) => b.savingsUSD - a.savingsUSD)
}

export function simulateModel(filter: QueryFilter, targetModel: string): SimulationResult {
  const records = queryRequests(filter)
  let currentCost = 0
  let simulatedCost = 0
  let highOutputCount = 0

  for (const r of records) {
    currentCost += r.costUSD
    const simCost = calculateCost(targetModel, {
      inputTokens: r.inputTokens,
      outputTokens: r.outputTokens,
      cacheReadTokens: r.cacheReadTokens,
      cacheWriteTokens: r.cacheWriteTokens,
    })
    simulatedCost += simCost
    if (r.outputTokens > 2000) highOutputCount++
  }

  const savings = currentCost - simulatedCost
  return {
    targetModel,
    currentCost,
    simulatedCost,
    savings: Math.max(0, savings),
    savingsPercent: currentCost > 0 ? (Math.max(0, savings) / currentCost) * 100 : 0,
    requestCount: records.length,
    warning: highOutputCount > 0 ? `${highOutputCount} complex requests (>2000 output tokens) may degrade with ${targetModel}` : undefined,
  }
}
```

```typescript
import chalk from 'chalk'
import { formatCost, header, dim } from './format.js'
import type { OptimizationPlan, SimulationResult } from '../analyzer/optimizer.js'

export function renderOptimizations(plans: OptimizationPlan[], totalCost: number): string {
  const lines = ['', header(`🔥 Total spending in period: ${formatCost(totalCost)}`), '']
  if (plans.length === 0) {
    lines.push(chalk.green('  ✨ No optimization opportunities found!'), '')
    return lines.join('\n')
  }

  const totalSavings = plans.reduce((s, p) => s + p.savingsUSD, 0)
  const totalPct = totalCost > 0 ? Math.round((totalSavings / totalCost) * 100) : 0
  lines.push(header(`  Optimization plans (${plans.length}, est. savings ${formatCost(totalSavings)}/period):`), '')

  for (let i = 0; i < plans.length; i++) {
    const p = plans[i]
    lines.push(chalk.bold(`  ${i + 1}. ${p.description}`) + dim(` — save ${formatCost(p.savingsUSD)}/period (${p.savingsPercent.toFixed(0)}%)`))
    if (p.fix) lines.push(dim(`     Fix: ${p.fix.split('\n')[0]}`))
    lines.push('')
  }

  lines.push('  ' + '─'.repeat(40))
  lines.push(chalk.bold.green(`  💰 Total savings: ${formatCost(totalSavings)}/period (${totalPct}%)`))
  lines.push(dim(`  📊 Current: ${formatCost(totalCost)} → Optimized: ${formatCost(totalCost - totalSavings)}`))
  lines.push('')
  return lines.join('\n')
}

export function renderSimulation(result: SimulationResult): string {
  const lines = [
    '',
    header(`  What-if: All requests used ${result.targetModel}`),
    '',
    `  Current:     ${formatCost(result.currentCost)}`,
    `  Simulated:   ${formatCost(result.simulatedCost)} (${result.savingsPercent.toFixed(0)}% ↓)`,
    `  Savings:     ${chalk.green(formatCost(result.savings))}`,
    `  Requests:    ${result.requestCount}`,
    '',
  ]
  if (result.warning) {
    lines.push(chalk.yellow(`  ⚠ ${result.warning}`))
    lines.push('')
  }
  return lines.join('\n')
}
```

```typescript
import { Command } from 'commander'
import { getDb } from '../db/db.js'
import { summarize } from '../analyzer/analyzer.js'
import { generateOptimizations, simulateModel } from '../analyzer/optimizer.js'
import { renderOptimizations, renderSimulation } from '../reporter/optimize.js'
import { applyFixToClaudeMd } from '../analyzer/fix-applier.js'
import chalk from 'chalk'

function parsePeriod(period: string): number {
  const match = period.match(/^(\d+(?:\.\d+)?)([hdwm])$/)
  if (!match) return Date.now() - 7 * 86400000
  const value = parseFloat(match[1])
  const ms: Record<string, number> = { h: 3600000, d: 86400000, w: 604800000, m: 2592000000 }
  return Date.now() - Math.round(value * ms[match[2]])
}

export const optimizeCommand = new Command('optimize')
  .description('Generate optimization plans to reduce AI spending')
  .option('--last <period>', 'Time period', '7d')
  .option('--simulate', 'Simulate a global model switch', false)
  .option('--model <model>', 'Target model for simulation', 'claude-haiku-4-5')
  .option('--apply', 'Auto-apply all fixes to CLAUDE.md', false)
  .option('--json', 'Output as JSON', false)
  .action((opts: { last: string; simulate: boolean; model: string; apply: boolean; json: boolean }) => {
    getDb()
    const since = parsePeriod(opts.last)
    const filter = { since }

    if (opts.simulate) {
      const result = simulateModel(filter, opts.model)
      if (opts.json) { console.log(JSON.stringify(result, null, 2)); return }
      process.stdout.write(renderSimulation(result))
      return
    }

    const plans = generateOptimizations(filter)
    const summary = summarize(filter)

    if (opts.json) { console.log(JSON.stringify({ totalCost: summary.totalCost, plans }, null, 2)); return }

    process.stdout.write(renderOptimizations(plans, summary.totalCost))

    if (opts.apply) {
      let applied = 0
      for (const p of plans) {
        if (p.fix) {
          const ok = applyFixToClaudeMd(p.fix)
          if (ok) { console.log(chalk.green(`  ✓ Applied: ${p.name}`)); applied++ }
        }
      }
      if (applied > 0) {
        console.log(chalk.bold.green(`\n  ${applied} fix(es) applied. Review: git diff CLAUDE.md\n`))
      }
    }
  })
```
