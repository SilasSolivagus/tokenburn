> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add subagent cost tree visualization (`tokenburn tree`), plan value calculation (`tokenburn report --plan`), and session-level aggregation (`tokenburn report --by session`) — features that differentiate tokenburn from both ccusage and BurnRate.

**Architecture:** The tree builder parses `parentUuid` chains from imported JSONL data in the `agent_tree` table to reconstruct agent hierarchies. Plan value compares actual usage cost against a subscription price. Session aggregation groups requests by `sessionId` (from logs) or by time gaps (from proxy data).

**Tech Stack:** TypeScript, better-sqlite3 (existing), chalk (existing)

```typescript
import { getDb } from '../db/db.js'

export interface AgentNode {
  uuid: string
  parentUuid: string | null
  costUSD: number
  inputTokens: number
  outputTokens: number
  toolUse: string[]
  children: AgentNode[]
}

export interface AgentTree {
  sessionId: string
  totalCost: number
  totalInputTokens: number
  totalOutputTokens: number
  nodeCount: number
  roots: AgentNode[]
  children: AgentNode[]  // alias for roots (backward compat)
  toolSummary: string
  durationMs: number
}

interface TreeRow {
  uuid: string
  parentUuid: string | null
  inputTokens: number
  outputTokens: number
  costUSD: number
  toolUse: string
  timestamp: number
}

export function buildAgentTree(sessionId: string): AgentTree | null {
  const db = getDb()
  const rows = db.prepare(
    'SELECT uuid, parentUuid, inputTokens, outputTokens, costUSD, toolUse, timestamp FROM agent_tree WHERE sessionId = ? ORDER BY timestamp ASC'
  ).all(sessionId) as TreeRow[]

  if (rows.length === 0) return null

  // Build node map
  const nodeMap = new Map<string, AgentNode>()
  for (const row of rows) {
    const tools: string[] = JSON.parse(row.toolUse)
    nodeMap.set(row.uuid, {
      uuid: row.uuid,
      parentUuid: row.parentUuid,
      costUSD: row.costUSD,
      inputTokens: row.inputTokens,
      outputTokens: row.outputTokens,
      toolUse: tools,
      children: [],
    })
  }

  // Link children to parents
  const roots: AgentNode[] = []
  for (const node of nodeMap.values()) {
    if (node.parentUuid && nodeMap.has(node.parentUuid)) {
      nodeMap.get(node.parentUuid)!.children.push(node)
    } else {
      roots.push(node)
    }
  }

  // Aggregate
  let totalCost = 0
  let totalInput = 0
  let totalOutput = 0
  const allTools: string[] = []
  for (const row of rows) {
    totalCost += row.costUSD
    totalInput += row.inputTokens
    totalOutput += row.outputTokens
    allTools.push(...JSON.parse(row.toolUse))
  }

  // Tool summary: count occurrences
  const toolCounts = new Map<string, number>()
  for (const t of allTools) {
    toolCounts.set(t, (toolCounts.get(t) ?? 0) + 1)
  }
  const toolSummary = Array.from(toolCounts.entries())
    .sort((a, b) => b[1] - a[1])
    .map(([name, count]) => `${name} ×${count}`)
    .join(', ')

  const timestamps = rows.map(r => r.timestamp)
  const durationMs = timestamps.length > 1 ? Math.max(...timestamps) - Math.min(...timestamps) : 0

  return {
    sessionId,
    totalCost,
    totalInputTokens: totalInput,
    totalOutputTokens: totalOutput,
    nodeCount: rows.length,
    roots,
    children: roots,
    toolSummary,
    durationMs,
  }
}

/**
 * Get recent sessions that have agent tree data.
 */
export function getTreeSessions(limit: number = 10): string[] {
  const db = getDb()
  const rows = db.prepare(
    'SELECT DISTINCT sessionId FROM agent_tree ORDER BY timestamp DESC LIMIT ?'
  ).all(limit) as Array<{ sessionId: string }>
  return rows.map(r => r.sessionId)
}
```

```typescript
import chalk from 'chalk'
import { formatCost, formatTokens, formatDuration, header, dim } from './format.js'
import type { AgentTree, AgentNode } from '../logs/tree-builder.js'

export function renderTree(tree: AgentTree): string {
  const lines = [
    '',
    header(`🌳 Session: ${tree.sessionId}`),
    dim(`   Cost: ${formatCost(tree.totalCost)} | ${tree.nodeCount} messages | ${formatDuration(tree.durationMs)}`),
    dim(`   Tools: ${tree.toolSummary || 'none'}`),
    '',
  ]

  for (let i = 0; i < tree.roots.length; i++) {
    const isLast = i === tree.roots.length - 1
    renderNode(tree.roots[i], '', isLast, lines)
  }

  lines.push('')
  return lines.join('\n')
}

function renderNode(node: AgentNode, prefix: string, isLast: boolean, lines: string[]): void {
  const connector = isLast ? '└── ' : '├── '
  const toolStr = node.toolUse.length > 0 ? ` [${node.toolUse.join(', ')}]` : ''
  const costStr = formatCost(node.costUSD)
  const tokenStr = dim(`(in: ${formatTokens(node.inputTokens)}, out: ${formatTokens(node.outputTokens)})`)

  lines.push(`${prefix}${connector}${costStr}${toolStr} ${tokenStr}`)

  const childPrefix = prefix + (isLast ? '    ' : '│   ')
  for (let i = 0; i < node.children.length; i++) {
    renderNode(node.children[i], childPrefix, i === node.children.length - 1, lines)
  }
}

export function renderTreeList(sessionIds: string[]): string {
  if (sessionIds.length === 0) {
    return '\n' + dim('  No agent tree data found. Import Claude Code logs first: tokenburn import') + '\n\n'
  }
  const lines = ['', header('  Recent sessions with agent data:'), '']
  for (let i = 0; i < sessionIds.length; i++) {
    lines.push(`  ${i + 1}. ${sessionIds[i]}`)
  }
  lines.push('', dim('  Run: tokenburn tree --session <id>'), '')
  return lines.join('\n')
}
```

```typescript
export const treeCommand = new Command('tree')

  .option('--last <n>', 'Show last N sessions', '1')
  .option('--json', 'Output as JSON', false)

    if (opts.session) {

      if (!tree) {
        console.log(`\n  No agent tree data for session: ${opts.session}\n`)
        return
      }
      if (opts.json) { console.log(JSON.stringify(tree, null, 2)); return }

      return

    if (sessions.length === 0) {

      return
    }

    if (opts.json) {
      const trees = sessions.map(id => buildAgentTree(id)).filter(Boolean)
      console.log(JSON.stringify(trees, null, 2))
      return
    }

    for (const id of sessions) {
      const tree = buildAgentTree(id)
      if (tree) process.stdout.write(renderTree(tree))
    }
  })
```

```typescript
export interface SessionAgg {
  sessionId: string
  totalCost: number
  totalInput: number
  totalOutput: number
  requestCount: number
  startTime: number
  endTime: number
  durationMs: number
}

export function aggregateBySession(filter: QueryFilter): SessionAgg[] {
  const db = getDb()
  const { sql, params } = buildWhere(filter)

  // Sessions with explicit sessionId
  const namedSessions = db.prepare(`
    SELECT sessionId, SUM(costUSD) AS totalCost, SUM(inputTokens) AS totalInput,
           SUM(outputTokens) AS totalOutput, COUNT(*) AS requestCount,
           MIN(timestamp) AS startTime, MAX(timestamp) AS endTime
    FROM requests ${sql} ${sql ? 'AND' : 'WHERE'} sessionId != ''
    GROUP BY sessionId
  `).all(params) as Array<SessionAgg & { startTime: number; endTime: number }>

  // Proxy data without sessionId — group by time gaps (>5 min = new session)
  const orphanRows = db.prepare(`
    SELECT timestamp, costUSD, inputTokens, outputTokens
    FROM requests ${sql} ${sql ? 'AND' : 'WHERE'} sessionId = ''
    ORDER BY timestamp ASC
  `).all(params) as Array<{ timestamp: number; costUSD: number; inputTokens: number; outputTokens: number }>

  const autoSessions: SessionAgg[] = []
  if (orphanRows.length > 0) {
    let current: { start: number; end: number; cost: number; input: number; output: number; count: number } = {
      start: orphanRows[0].timestamp, end: orphanRows[0].timestamp,
      cost: orphanRows[0].costUSD, input: orphanRows[0].inputTokens,
      output: orphanRows[0].outputTokens, count: 1,
    }

    for (let i = 1; i < orphanRows.length; i++) {
      const row = orphanRows[i]
      if (row.timestamp - current.end > 300000) {  // 5 min gap
        autoSessions.push({
          sessionId: `auto-${current.start}`,
          totalCost: current.cost, totalInput: current.input, totalOutput: current.output,
          requestCount: current.count, startTime: current.start, endTime: current.end,
          durationMs: current.end - current.start,
        })
        current = { start: row.timestamp, end: row.timestamp, cost: row.costUSD, input: row.inputTokens, output: row.outputTokens, count: 1 }
      } else {
        current.end = row.timestamp
        current.cost += row.costUSD
        current.input += row.inputTokens
        current.output += row.outputTokens
        current.count++
      }
    }
    autoSessions.push({
      sessionId: `auto-${current.start}`,
      totalCost: current.cost, totalInput: current.input, totalOutput: current.output,
      requestCount: current.count, startTime: current.start, endTime: current.end,
      durationMs: current.end - current.start,
    })
  }

  const allSessions = [
    ...namedSessions.map(s => ({ ...s, durationMs: s.endTime - s.startTime })),
    ...autoSessions,
  ]

  return allSessions.sort((a, b) => b.totalCost - a.totalCost)
}
```

```typescript
export interface PlanValue {
  planPriceUSD: number
  apiEquivalentUSD: number
  multiplier: number
  verdict: string
}

export function calculatePlanValue(filter: QueryFilter, planPriceUSD: number): PlanValue {
  const summary = summarize(filter)
  const apiEquivalentUSD = summary.totalCost
  const multiplier = planPriceUSD > 0 ? apiEquivalentUSD / planPriceUSD : 0

  let verdict: string
  if (multiplier >= 2) {
    verdict = `Great value! You'd pay ${multiplier.toFixed(1)}x more on API pricing.`
  } else if (multiplier >= 1) {
    verdict = `Fair value. API equivalent is ${multiplier.toFixed(1)}x your plan price.`
  } else if (multiplier > 0) {
    verdict = `Underusing your plan. Consider a lower tier to save $${(planPriceUSD - apiEquivalentUSD).toFixed(0)}/month.`
  } else {
    verdict = 'No usage data to evaluate.'
  }

  return { planPriceUSD, apiEquivalentUSD, multiplier, verdict }
}
```
